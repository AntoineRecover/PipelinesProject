pipeline {
    agent any
    environment {
        GitBranch = "main"
        EC2_IP = "$EC2IP"
    }
    stages {
        stage ("Clean Jenkins workspace before start") {
            steps {
                cleanWs()
            }
        }

        stage ("checkout_git") {
            steps {
                sh "pwd"
                dir("Build_WebAMI") {
                    sh "pwd"
                    git(
                        url: 'git@github.com:nsxfrtn/Projet_DevOps.git',
                        credentialsId: "$GitCredentials",
                        branch: "${GitBranch}"
                    )
                    sh "ls"
                }
            }
        }

        stage ("Build AMI") {
            steps {
                dir("Build_WebAMI") {
                    dir("Packer") {
                        sh "pwd"
                        sh "ls"
                        
                        wrap([$class: 'AnsiColorBuildWrapper', 'colorMapName': 'xterm']){
                            sh "packer init ."
                            sh "packer fmt ."
                            sh "packer validate ."
                            sh "packer build -var region=$region -var subnet_id=$subnet_id -var ami_name=$ami_name -var port=$port -var pcks=$pcks . "
                        }
                    }
                }
            }
        }
    }
}
